import {
  Canvas,
  Meta,
  Story,
  ArgsTable
} from '@storybook/addon-docs'
import RplDescriptionList from './description-list.vue'
import { a11yStoryCheck } from './../../../stories/interactions.js'

export const SingleTemplate = (args) => ({
  components: { RplDescriptionList },
  setup() {
    return { args }
  },
  template: '<RplDescriptionList v-bind="args" />'
})

export const InlineTemplate = (args) => ({
  components: { RplDescriptionList },
  setup() {
    return { args }
  },
  template: '<div class="rpl-storybook__page"><RplDescriptionList v-for="(list, i) in args.rows" :items="[list]" :index="i" class="rpl-description-list--inline" /></div>'
})

<Meta
  title='Core/Containers/Description list'
  component={RplDescriptionList}
/>

## Inline

Use multiple `RplDescriptionList` components with the `rpl-description-list--inline` variant class to display the term inline.

<Canvas>
  <Story
